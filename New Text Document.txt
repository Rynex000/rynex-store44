    <script>
        const MASTER_PASS = "rynex123";

        function checkAdmin() {
            if(document.getElementById('adminPass').value === MASTER_PASS) {
                document.getElementById('adminLogin').style.display = 'none';
                document.getElementById('adminPanel').style.opacity = '1';
                loadStoreData();
                loadOrders();
            } else {
                document.getElementById('loginErr').classList.remove('hidden');
            }
        }

        function loadStoreData() {
            // Load standard prices
            document.getElementById('price_ai_mo').value = localStorage.getItem('price_ai_mo') || "7.49";
            document.getElementById('price_ai_yr').value = localStorage.getItem('price_ai_yr') || "70.00";
            document.getElementById('price_mov_mo').value = localStorage.getItem('price_mov_mo') || "9.99";
            document.getElementById('price_mov_yr').value = localStorage.getItem('price_mov_yr') || "110.00";
            document.getElementById('price_learn_mo').value = localStorage.getItem('price_learn_mo') || "15.00";
            document.getElementById('price_learn_yr').value = localStorage.getItem('price_learn_yr') || "150.00";
            
            // Load special offer data
            document.getElementById('offer_name').value = localStorage.getItem('offer_name') || "Summer Mega Offer";
            document.getElementById('offer_price').value = localStorage.getItem('offer_price') || "49.99";
        }

        function saveStoreData() {
            localStorage.setItem('price_ai_mo', document.getElementById('price_ai_mo').value);
            localStorage.setItem('price_ai_yr', document.getElementById('price_ai_yr').value);
            localStorage.setItem('price_mov_mo', document.getElementById('price_mov_mo').value);
            localStorage.setItem('price_mov_yr', document.getElementById('price_mov_yr').value);
            localStorage.setItem('price_learn_mo', document.getElementById('price_learn_mo').value);
            localStorage.setItem('price_learn_yr', document.getElementById('price_learn_yr').value);
            
            // Save special offer
            localStorage.setItem('offer_name', document.getElementById('offer_name').value);
            localStorage.setItem('offer_price', document.getElementById('offer_price').value);
            
            alert("All prices and special offers updated successfully!");
        }

        function loadOrders() {
            const orders = JSON.parse(localStorage.getItem('rynex_all_orders')) || [];
            const list = document.getElementById('adminOrderList');
            list.innerHTML = orders.length === 0 ? '<tr><td colspan="5" class="p-10 text-center text-slate-600 uppercase font-black text-[10px]">No orders available</td></tr>' : "";

            orders.reverse().forEach((order, index) => {
                const originalIndex = orders.length - 1 - index;
                list.innerHTML += `
                    <tr class="hover:bg-white/[0.02] transition">
                        <td class="p-5">
                            <div class="font-bold text-blue-400 text-xs">${order.email}</div>
                            <div class="text-[9px] text-slate-500">${order.date}</div>
                        </td>
                        <td class="p-5 font-black text-slate-300 uppercase text-[10px] italic">${order.service}</td>
                        <td class="p-5 font-black text-white">$${order.price}</td>
                        <td class="p-5">
                            <span class="${order.status === 'Completed' ? 'text-green-500' : 'text-orange-500'} font-black uppercase text-[9px]">
                                ${order.status}
                            </span>
                        </td>
                        <td class="p-5 text-center">
                            ${order.status === 'Pending' ? 
                                `<button onclick="completeOrder(${originalIndex})" class="bg-blue-600 text-white px-4 py-1.5 rounded-lg font-black text-[9px] uppercase hover:bg-blue-500 transition">Complete</button>` 
                                : `<span class="text-green-500 font-black text-[9px] uppercase">Finished</span>`}
                        </td>
                    </tr>
                `;
            });
        }

        function completeOrder(idx) {
            let orders = JSON.parse(localStorage.getItem('rynex_all_orders'));
            orders[idx].status = "Completed";
            localStorage.setItem('rynex_all_orders', JSON.stringify(orders));
            loadOrders();
        }
    </script>
